   def subtraction(x: int) -> int:
        if not x == 10 or x > 8:
            while x > 7: 
                x -= 1
        return x

    def to_ten(y: int) -> int:
        while y < 10:
            y += 2
        return y

    def main():
        num1: int = 2
        num1 = to_ten(num1)
        num1 = subtraction(num1)
        print(f"The number is {num1}")

    if __name__ == "__main__":
        main()
